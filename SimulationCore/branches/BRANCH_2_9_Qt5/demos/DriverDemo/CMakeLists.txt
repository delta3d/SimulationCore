SET(LIB_NAME   DriverDemo)

SET(HEADER_PATH include)
SET(SOURCE_PATH source)

INCLUDE_DIRECTORIES(${HEADER_PATH})

file(GLOB LIB_PUBLIC_HEADERS 
   "${HEADER_PATH}/*.h"
   )

file(GLOB LIB_SOURCES 
   "${SOURCE_PATH}/*.cpp"
   )


set(LIB_EXTERNAL_DEPS
   DTNETGM_LIBRARY
)

set(LIB_DEPS
   ${SIMCORE_LIBRARY}
)

if (BUILD_HLA)
   ADD_DEFINITIONS(-DBUILD_HLA)
   list(APPEND LIB_DEPS 
      ${SIMCOREHLA_LIBRARY}
   )
endif (BUILD_HLA)

DELTA3D_ADD_LIBRARY(${LIB_NAME} DRIVER_DEMO_LIBRARY SHARED)


BUILD_GAME_START(${LIB_NAME} ON)
